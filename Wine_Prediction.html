scores={}
scores_list=[]

for k in k_range:
    knn_cv = KNeighborsClassifier(n_neighbors=k)
    cv_scores = cross_val_score(knn_cv, scaled_features, Y, cv=10)
    scores[k]= np.mean(cv_scores)
    scores_list.append(np.mean(cv_scores))
